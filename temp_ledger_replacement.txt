                                    {entry.policy.policyNumber}
                                  </Link>
                                ) : (

                                )}
                              </td>
                              <td className="px-4 py-3 text-sm text-right font-medium text-red-600">

                                {entry.entryType === 'CREDIT' ? formatCurrency(amount) : '-'}
                              </td>

                                <span className={runningBalance < 0 ? 'text-red-600' : 'text-green-600'}>

                    )}
                  </tbody>
                </table>
              </div>
            </CardContent>
